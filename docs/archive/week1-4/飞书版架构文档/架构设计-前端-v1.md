## 1. 技术栈总览

|------|---------|------|

| 语言 | TypeScript | 严格模式，`vue-tsc --noEmit` 验证 |

### 3.2 View 标准结构

```vue
<script setup lang="ts">
import { computed } from 'vue'
import { useFilterStore } from '@/stores/filterStore'
import { useQuery } from '@tanstack/vue-query'
import { useApi } from '@/api/useApi'

// ① Filter Store
const filterStore = useFilterStore()

// ② API Query（固定写法）
const { data, isLoading, error } = useQuery(
  useApi('/api/v1/xxx/endpoint', {
    query: computed(() => ({
      ...filterStore.queryParams,
      lookback_days: 90,
      metric_type: 'GMV'
    }))
  })
)

// ③ Computed 处理
const kpis = computed(() => data.value?.kpis ?? [])
</script>

<template>
  <DefaultLayout>
    <PageHeader title="页面标题" subtitle="副标题" />
    <!-- KPI Cards -->
    <div class="grid grid-cols-4 gap-4">
      <MetricCard v-for="kpi in kpis" :key="kpi.label" v-bind="kpi" />
    </div>
    <!-- Main Content -->
    <NTabs>...</NTabs>
  </DefaultLayout>
</template>
```

> 📌 **Query Key 规范**：必须传入 `ComputedRef`，由 Vue Query v5 自动解包。禁止传入函数（会导致每次渲染生成新函数引用，缓存失效）。

## 5. 状态管理：Pinia FilterStore

`filterStore` 是全局筛选状态的唯一管理点：

```typescript
// stores/filterStore.ts
export const useFilterStore = defineStore('filter', () => {
  const dateRange = ref<string[]>([startOfWeek(), endOfWeek()]) // 默认 WTD
  const channel = ref<string>('全店')
  const segment = ref<string>('全店')
  const lookbackDays = ref<number>(90)
  const metricType = ref<'GMV' | 'GSV'>('GMV')

  // 暴露给 URL 的查询参数
  const queryParams = computed(() => ({
    start_date: dateRange.value[0],
    end_date: dateRange.value[1],
    channel: channel.value === '全店' ? undefined : channel.value,
    segment: segment.value === '全店' ? undefined : segment.value,
    lookback_days: lookbackDays.value,
    metric_type: metricType.value,
  }))

  return { dateRange, channel, segment, lookbackDays, metricType, queryParams }
})
```

> 📌 **URL 同步**：`composables/useFilterSync.ts` 负责 Pinia ↔ URL 的双向同步，使用 `watch(flush: 'post')` 避免路由守卫死循环。
