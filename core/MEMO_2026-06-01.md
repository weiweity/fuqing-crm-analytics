```python
def _check_business_smoothness(csv_file, data, threshold=0.30):
    prev_total = get_previous_day_total(csv_file, data['item_id'], prev_date_str)
    change_ratio = (current_total - prev_total) / prev_total
    if abs(change_ratio) > threshold:
        return f"资产总量从 {prev_total:,} {方向}到 {current_total:,} ({change_ratio*100:+.1f}%)"
```
